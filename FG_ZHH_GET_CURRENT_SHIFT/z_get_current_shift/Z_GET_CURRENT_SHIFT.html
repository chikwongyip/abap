FUNCTION Z_GET_CURRENT_SHIFT.
*"----------------------------------------------------------------------
*"*"Local Interface:
*" CHANGING
*" REFERENCE(SHIFT) TYPE C OPTIONAL
*"----------------------------------------------------------------------
* night shift 000000 - 075959
* morning shift 080000 - 155959
* noon shift 160000 - 235959

* Global data declarations

  DATA:v_uzeit LIKE sy-uzeit.
  v_uzeit = sy-uzeit.
  CLEAR: shift.
* night shift 000000 - 075959
  IF v_uzeit >= '000000' AND v_uzeit <= '075959'.
    shift = '3'.
  ENDIF.

  IF v_uzeit >= '080000' AND v_uzeit <= '155959'.
    shift = '1'.
  ENDIF.

  IF v_uzeit >= '160000' AND v_uzeit <= '235959'.
    shift = '2'.
  ENDIF.

ENDFUNCTION.
