FUNCTION Z_CHECK_AUTHOR_SIMPLE.
*"----------------------------------------------------------------------
*"*"Local Interface:
*" IMPORTING
*" REFERENCE(OBJECT) TYPE XUOBJECT
*" REFERENCE(FIELD) TYPE XUFIELD
*" REFERENCE(VALUE) TYPE ANY
*"----------------------------------------------------------------------

* Global data declarations

  DATA:wa_tobj  TYPE tobj.
  DATA:wa_tobjt TYPE tobjt.
  DATA:message TYPE string.
  CLEAR:wa_tobj,wa_tobjt.
  SELECT SINGLE * INTO CORRESPONDING FIELDS OF wa_tobj FROM tobj WHERE objct = object
                                                                   AND (
                                                                           fiel1 = field
                                                                         OR
                                                                           fiel2 = field
                                                                         OR
                                                                           fiel3 = field
                                                                         OR
                                                                           fiel4 = field
                                                                         OR
                                                                           fiel5 = field
                                                                         OR
                                                                           fiel6 = field
                                                                         OR
                                                                           fiel7 = field
                                                                         OR
                                                                           fiel8 = field
                                                                         OR
                                                                           fiel9 = field
                                                                         OR
                                                                           fiel0 = field
                                                                        ).
    IF sy-subrc = 0.
      AUTHORITY-CHECK OBJECT object
                          ID field   FIELD value
                          ID 'ACTVT' DUMMY.
      IF sy-subrc <> 0.
        SELECT SINGLE * INTO CORRESPONDING FIELDS OF wa_tobjt  FROM tobjt WHERE langu  = sy-langu
                                                                            AND object = object.

        CONCATENATE 'You do not have author' wa_tobjt-ttext 'for field:' value
               INTO message
       SEPARATED BY space.
       MESSAGE message TYPE 'E'.
      ENDIF.
    ELSE.
      MESSAGE 'Not exist author object:' && object TYPE 'I'.
    ENDIF.



ENDFUNCTION.
