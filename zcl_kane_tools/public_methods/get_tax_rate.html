  method GET_TAX_RATE.
    DATA:wa_a003 TYPE a003.
    DATA:wa_konp TYPE konp.
    CLEAR:wa_a003,wa_konp.
    SELECT SINGLE *
      INTO CORRESPONDING FIELDS OF wa_a003
      FROM a003
      WHERE a003~kappl = 'TX'
        AND a003~kschl = tax_type
        AND a003~aland = country
        AND a003~mwskz = tax_code.

    SELECT SINGLE *
      INTO CORRESPONDING FIELDS OF wa_konp
      FROM konp
      WHERE konp~knumh = wa_a003-knumh
        AND konp~kschl = wa_a003-kschl.
    tax_rate = wa_konp-kbetr / 1000.
  endmethod.
