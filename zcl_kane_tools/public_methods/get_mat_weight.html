Description: get material gross weight & net weight by material && unit

*   Method attributes.                                                   *

Instantiation: Public

  method GET_MAT_WEIGHT.
    DATA:wa_mara TYPE mara,
         umrez   TYPE umrez,
         umren   TYPE umren.
    CLEAR:wa_mara,umrez,umren.

    SELECT SINGLE *
      INTO CORRESPONDING FIELDS OF wa_mara
      FROM mara
      WHERE matnr = matnr.
    CALL METHOD me->get_conversion
      EXPORTING
        matnr = matnr
        from_unit = wa_mara-meins
        to_unit   = vrkme
      IMPORTING
        umrez     =  umrez
        umren     =  umren.
    gross_weight = ( umrez / umren ) * wa_mara-brgew.
    net_weight   = ( umrez / umren ) * wa_mara-ntgew.

  endmethod.
