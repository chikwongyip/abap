  method CONVERT_DATE_FORMAT.
    DATA:istring TYPE char20.
    DATA:icount  TYPE i.
    istring = date.
    IF language = 'E'.
      DATA(month) = zcl_kane_tools=>mapping_month( month = istring+4(2) ).
    ELSE.
      month = date+4(2).
    ENDIF.
    CONCATENATE istring+0(4) month istring+6(2) INTO date_string SEPARATED BY  date_seprate.
    IF time IS NOT INITIAL.
      istring = time.
      CONCATENATE istring+0(2) istring+2(2) istring+4(2) INTO istring SEPARATED BY time_seprate.
      CONCATENATE date_string istring INTO date_string SEPARATED BY space.
    ENDIF.
  endmethod.
