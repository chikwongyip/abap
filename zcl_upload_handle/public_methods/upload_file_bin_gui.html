METHOD upload_file_bin_gui.
  DATA:wa_directory_bin TYPE zdirectory_bin.
  LOOP AT i_directory INTO DATA(wa_directory).
    CLEAR:wa_directory_bin.
    wa_directory_bin-filename   = wa_directory-name.
    CONCATENATE i_path '\' wa_directory-name INTO wa_directory_bin-fullname.
     wa_directory_bin-size      = wa_directory-size.
     wa_directory_bin-rc        = wa_directory-rc.
     wa_directory_bin-filetype  = wa_directory-filetype.
     CALL FUNCTION 'GUI_UPLOAD'
       EXPORTING
         filename               = wa_directory_bin-fullname
         filetype               = 'BIN'
       TABLES
         data_tab               = wa_directory_bin-bin
       EXCEPTIONS OTHERS        = 1.
     APPEND wa_directory_bin TO e_directory_bin.
  ENDLOOP.
ENDMETHOD.
